### Database Schema

```sql
-- Time-series data (TimescaleDB)
CREATE TABLE profile_metrics (
  time TIMESTAMPTZ NOT NULL,
  username TEXT NOT NULL,
  followers INT,
  total_stars INT,
  total_repos INT,
  profile_views INT, -- estimated
  PRIMARY KEY (time, username)
);

SELECT create_hypertable('profile_metrics', 'time');

-- Repository metrics
CREATE TABLE repo_metrics (
  time TIMESTAMPTZ NOT NULL,
  repo_full_name TEXT NOT NULL,
  stars INT,
  forks INT,
  watchers INT,
  issues_open INT,
  PRIMARY KEY (time, repo_full_name)
);

-- User settings
CREATE TABLE users (
  id UUID PRIMARY KEY,
  github_username TEXT UNIQUE,
  email TEXT,
  tier TEXT DEFAULT 'free', -- free, pro, team
  created_at TIMESTAMPTZ,
  last_sync TIMESTAMPTZ
);

-- Competitor tracking (premium feature)
CREATE TABLE competitor_tracking (
  user_id UUID REFERENCES users(id),
  competitor_username TEXT,
  added_at TIMESTAMPTZ,
  PRIMARY KEY (user_id, competitor_username)
);
```
