// Run on 1st of each month
async function processMonthlyPayouts() {
  const sellers = await db.query(`
    SELECT 
      seller_id,
      SUM(amount * 0.70) as earnings
    FROM purchases
    WHERE purchased_at >= date_trunc('month', CURRENT_DATE - INTERVAL '1 month')
      AND purchased_at < date_trunc('month', CURRENT_DATE)
    GROUP BY seller_id
    HAVING SUM(amount * 0.70) >= 10.00 -- Minimum payout $10
  `);
  
  for (const seller of sellers) {
    // Transfer via Stripe Connect
    const transfer = await stripe.transfers.create({
      amount: seller.earnings * 100, // Convert to cents
      currency: 'usd',
      destination: seller.stripe_account_id,
    });
    
    // Record payout
    await db.insert('payouts', {
      seller_id: seller.id,
      amount: seller.earnings,
      stripe_transfer_id: transfer.id,
      status: 'completed',
    });
    
    // Email seller
